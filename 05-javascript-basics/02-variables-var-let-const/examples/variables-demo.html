Scope Comparison

var — Function Scoped (leaks)

if (true) {
  var x = 'I leak out';
}
console.log(x); // works — leaks!

let — Block Scoped (contained)

if (true) {
  let y = 'I stay inside';
  console.log(y); // works
}
console.log(y); // ReferenceError

const — Cannot Reassign

const MAX = 100;
MAX = 200; // TypeError

// But mutation is OK:
const user = { age: 25 };
user.age = 26; // ✅ works

Naming Conventions

// camelCase for variables
const firstName = 'Alex';
let totalScore = 0;

// SCREAMING_SNAKE_CASE for true constants
const MAX_RETRIES = 3;
const API_BASE_URL = 'https://api.example.com';

// PascalCase for classes
class UserProfile {}

// ❌ Avoid vague names
let x = 42;
let data = {};
let temp = 'hello';
